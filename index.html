const express = require("express"); const cors = require("cors"); require("dotenv").config(); const app = express(); app.use(cors()); // Endpoint that returns firebase config app.get("/firebase-config", (req, res) => { res.json({ apiKey: process.env.FIREBASE_API_KEY, authDomain: process.env.FIREBASE_AUTH_DOMAIN, projectId: process.env.FIREBASE_PROJECT_ID, storageBucket: process.env.FIREBASE_STORAGE_BUCKET, messagingSenderId: process.env.FIREBASE_MESSAGING_SENDER_ID, appId: process.env.FIREBASE_APP_ID, }); }); const PORT = process.env.PORT || 5000; app.listen(PORT, () => console.log(`âœ… Server running on port ${PORT}`));
